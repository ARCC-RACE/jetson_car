472  throw gnublin_smtp(gnublin_smtp::WSA_INVALID_SOCKET);

477  ReceiveData();
478  switch(SmtpXYZdigits())

491  bAccepted = false;

508  bAccepted = false;
509  do

569

588

590  if(!(rcpt_count = Recipients.size()))
591  throw gnublin_smtp(gnublin_smtp ::UNDEF_RECIPIENTS);
592  for(i=0;i<Recipients.size();i++)
593  {

595  SendData();
596  bAccepted = false;
597  do
598  {

616  SendData();
617  bAccepted = false;
618  do
619  {
620  ReceiveData();
621  switch(SmtpXYZdigits())
622  {
623  case 250:
624  bAccepted = true;
625  break;
626  default:
627  ; // not necessary to throw
628  }
629  }while(!bAccepted);
630  }
631 
632  for(i=0;i<BCCRecipients.size();i++)
633  {
634  sprintf(SendBuf,"RCPT TO:<%s>\r\n",(BCCRecipients.at(i).Mail).c_str());
635  SendData();
636  bAccepted = false;
637  do
638  {
639  ReceiveData();
640  switch(SmtpXYZdigits())
641  {
642  case 250:
643  bAccepted = true;
644  break;

646  ; // not necessary to throw
647  }
648  }while(!bAccepted);
649  }

652  strcpy(SendBuf,"DATA\r\n");
653  SendData();
654  bAccepted = false;
655  do
656  {
657  ReceiveData();
658  switch(SmtpXYZdigits())
659  {
660  case 354:
661  bAccepted = true;
662  break;
663  case 250:
664  break;
665  default:
666  throw gnublin_smtp(gnublin_smtp ::COMMAND_DATA);
667  }
668  }while(!bAccepted);
669 
670  // send header(s)
671  FormatHeader(SendBuf);
672  SendData();
673 
674  // send text message
675  if(GetMsgLines())
676  {
677  for(i=0;i<GetMsgLines();i++)
678  {
679  sprintf(SendBuf,"%s\r\n",GetMsgLineText(i));
680  SendData();
681  }
682  }
683  else
684  {
685  sprintf(SendBuf,"%s\r\n"," ");
686  SendData();
687  }
688 
689  // next goes attachments (if they are)
690  if((FileBuf = new char[55]) == NULL)
691  throw gnublin_smtp(gnublin_smtp ::LACK_OF_MEMORY);
692 
693  if((FileName = new char[255]) == NULL)
694  throw gnublin_smtp(gnublin_smtp ::LACK_OF_MEMORY);
695 
696  TotalSize = 0;
697  for(FileId=0;FileId<Attachments.size();FileId++)
698  {
699  strcpy(FileName,Attachments[FileId].c_str());
700 
701  sprintf(SendBuf,"--%s\r\n",BOUNDARY_TEXT);
702  strcat(SendBuf,"Content-Type: application/x-msdownload; name=\"");
703  strcat(SendBuf,&FileName[Attachments[FileId].find_last_of("\\") + 1]);
704  strcat(SendBuf,"\"\r\n");
705  strcat(SendBuf,"Content-Transfer-Encoding: base64\r\n");
706  strcat(SendBuf,"Content-Disposition: attachment; filename=\"");
707  strcat(SendBuf,&FileName[Attachments[FileId].find_last_of("\\") + 1]);
708  strcat(SendBuf,"\"\r\n");
709  strcat(SendBuf,"\r\n");
710 
711  SendData();
712

714  hFile = fopen(FileName,"rb");
715  if(hFile == NULL)
716  throw gnublin_smtp(gnublin_smtp ::FILE_NOT_EXIST);
717 
718  // checking file size:
719  FileSize = 0;
720  while(!feof(hFile))
721  FileSize += fread(FileBuf,sizeof(char),54,hFile);
722  TotalSize += FileSize;
723 
724  // sending the file:
725  if(TotalSize/1024 > MSG_SIZE_IN_MB*1024)
726  throw gnublin_smtp(gnublin_smtp ::MSG_TOO_BIG);
727  else
728  {
729  fseek (hFile,0,SEEK_SET);
730 
731  MsgPart = 0;
732  for(i=0;i<FileSize/54+1;i++)
733  {
734 
735  res = fread(FileBuf,sizeof(char),54,hFile);
736  MsgPart ? strcat(SendBuf,base64_encode(reinterpret_cast<const unsigned char*>(FileBuf),res).c_str())
737  : strcpy(SendBuf,base64_encode(reinterpret_cast<const unsigned char*>(FileBuf),res).c_str());
738  strcat(SendBuf,"\r\n");
739  MsgPart += res + 2;
740  if(MsgPart >= BUFFER_SIZE/2)
741  { // sending part of the message
742  MsgPart = 0;
743  SendData(); // FileBuf, FileName, fclose(hFile);
744  }
745  }
746  if(MsgPart)
747  {
748  SendData(); // FileBuf, FileName, fclose(hFile);
749  }
750  }
751  fclose(hFile);
752  }
753  delete[] FileBuf;
754  delete[] FileName;
755 
756  // sending last message block (if there is one or more attachments)
757  if(Attachments.size())
758  {
759  sprintf(SendBuf,"\r\n--%s--\r\n",BOUNDARY_TEXT);
760  SendData();
761  }
762

764  strcpy(SendBuf,"\r\n.\r\n");
765  SendData();
766  bAccepted = false;
767  do
768  {
769  ReceiveData();
770  switch(SmtpXYZdigits())
771  {
772  case 250:
773  bAccepted = true;
774  break;
775  default:
776  throw gnublin_smtp(gnublin_smtp ::MSG_BODY_ERROR);
777  }
778  }while(!bAccepted);
779

781

784  SendData();
785  bAccepted = false;
786  do
787  {
788  ReceiveData();
789  switch(SmtpXYZdigits())
790  {
791  case 221:
792  bAccepted = true;
793  break;
794  default:
795  throw gnublin_smtp(gnublin_smtp ::COMMAND_QUIT);
796  }
797  }while(!bAccepted);
798 
799

801  //hSocket = NULL;
802 }

832 SOCKET gnublin_smtp::ConnectRemoteServer(const char *szServer,const unsigned short nPort_)
833 {
834  unsigned short nPort = 0;
835  LPSERVENT lpServEnt;
836  SOCKADDR_IN sockAddr;
837  unsigned long ul = 1;
838  fd_set fdwrite,fdexcept;
839  timeval timeout;
840  int res = 0;
841 
842  timeout.tv_sec = TIME_IN_SEC;
843  timeout.tv_usec = 0;
844 
845  SOCKET hSocket = INVALID_SOCKET;
846 
847  if((hSocket = socket(PF_INET, SOCK_STREAM,0)) == INVALID_SOCKET)
848  throw gnublin_smtp(gnublin_smtp ::WSA_INVALID_SOCKET);
849 
850  if(nPort_ != 0)
851  nPort = htons(nPort_);
852  else
853  {
854  lpServEnt = getservbyname("mail", 0);
855  if (lpServEnt == NULL)
856  nPort = htons(25);
857  else
858  nPort = lpServEnt->s_port;
859  }
860 
861  sockAddr.sin_family = AF_INET;
862  sockAddr.sin_port = nPort;
863  if((sockAddr.sin_addr.s_addr = inet_addr(szServer)) == INADDR_NONE)
864  {
865  LPHOSTENT host;
866 
867  host = gethostbyname(szServer);
868  if (host)
869  memcpy(&sockAddr.sin_addr,host->h_addr_list[0],host->h_length);
870  else
871  {
872  close(hSocket);
873  throw gnublin_smtp(gnublin_smtp ::WSA_GETHOSTBY_NAME_ADDR);
874  }
875  }
876 
877  // start non-blocking mode for socket:
878  if(ioctl(hSocket,FIONBIO, (unsigned long*)&ul) == SOCKET_ERROR)
879  {
880  close(hSocket);
881  throw gnublin_smtp(gnublin_smtp ::WSA_IOCTLSOCKET);
882  }

1115  strcat(header, "Subject: ");

1117  }
