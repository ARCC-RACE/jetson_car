589 int gnublin_module_step::drive(int steps) {
590  int old_position;
591  int new_position;
592  bool steps_limit=false; //if steps > 32760 devide it in steps < 32760!
593  #define STEP_LIMIT 32760
594  #define STEP_LIMIT_WAIT_TIME 200
595  do{
596  old_position = getActualPosition();
597  if(abs(steps)>STEP_LIMIT){
598  steps_limit=true;
599  if(steps>0){
600  new_position = old_position + STEP_LIMIT;
601  steps-=STEP_LIMIT;
602  } else{
603  new_position = old_position - STEP_LIMIT;
604  steps+=STEP_LIMIT;
605  }
606  }
607  else{
608  steps_limit=false;
609  new_position = old_position + steps;
610  }
611  if(!setPosition(new_position)) {
612  return -1;
613  }
614  if(steps_limit){
615  while(getMotionStatus() != 0) {
616  usleep(STEP_LIMIT_WAIT_TIME);
617  }
618  }
619  }while(steps_limit);
620  return 0;
